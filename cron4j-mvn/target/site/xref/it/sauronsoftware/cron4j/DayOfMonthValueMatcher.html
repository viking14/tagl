19  package it.sauronsoftware.cron4j;
20  
21  import java.util.ArrayList;
22  
23  /**
24   * <p>
25   * A ValueMatcher whose rules are in a plain array of integer values. When asked
26   * to validate a value, this ValueMatcher checks if it is in the array and, if
27   * not, checks whether the last-day-of-month setting applies.
28   * </p>
29   * 
30   * @author Paul Fernley
31   */
32  class DayOfMonthValueMatcher extends IntArrayValueMatcher {
33  
34  	private static final int[] lastDays = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
35  
36  	/**
37  	 * Builds the ValueMatcher.
38  	 * 
39  	 * @param integers
40  	 *            An ArrayList of Integer elements, one for every value accepted
41  	 *            by the matcher. The match() method will return true only if
42  	 *            its parameter will be one of this list or the
43  	 *            last-day-of-month setting applies.
44  	 */
45  	public DayOfMonthValueMatcher(ArrayList integers) {
46  		super(integers);
47  	}
48  
49  	/**
50  	 * Returns true if the given value is included in the matcher list or the
51  	 * last-day-of-month setting applies.
52  	 */
53  	public boolean match(int value, int month, boolean isLeapYear) {
54  		return (super.match(value) || (value > 27 && match(32) && isLastDayOfMonth(value, month, isLeapYear)));
55  	}
56  
57  	public boolean isLastDayOfMonth(int value, int month, boolean isLeapYear) {
58  		if (isLeapYear && month == 2) {
59  			return value == 29;
60  		} else {
61  			return value == lastDays[month - 1];
62  		}
63  	}
64  
65  }
