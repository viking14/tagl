265 	/**
266 	 * Implements {@link Task#execute(TaskExecutionContext)}. Runs the given
267 	 * command as a separate process and waits for its end.
268 	 */
269 	public void execute(TaskExecutionContext context) throws RuntimeException {
270 		Process p;
271 		try {
272 			p = exec();
273 		} catch (IOException e) {
274 			throw new RuntimeException(toString() + " cannot be started", e);
275 		}
276 		InputStream in = buildInputStream(stdinFile);
277 		OutputStream out = buildOutputStream(stdoutFile);
278 		OutputStream err = buildOutputStream(stderrFile);
279 		if (in != null) {
280 			StreamBridge b = new StreamBridge(in, p.getOutputStream());
281 			b.start();
282 		}
283 		if (out != null) {
284 			StreamBridge b = new StreamBridge(p.getInputStream(), out);
285 			b.start();
286 		}
287 		if (err != null) {
288 			StreamBridge b = new StreamBridge(p.getErrorStream(), err);
289 			b.start();
290 		}
291 		int r;
292 		try {
293 			r = p.waitFor();
294 		} catch (InterruptedException e) {
295 			throw new RuntimeException(toString() + " has been interrupted");
296 		} finally {
297 			if (in != null) {
298 				try {
299 					in.close();
300 				} catch (Throwable e) {
301 				}
302 			}
303 			if (out != null) {
304 				try {
305 					out.close();
306 				} catch (Throwable e) {
307 				}
308 			}
309 			if (err != null) {
310 				try {
311 					err.close();
312 				} catch (Throwable e) {
313 				}
314 			}
315 			p.destroy();
316 		}
317 		if (r != 0) {
318 			throw new RuntimeException(toString() + " returns with error code " + r);
319 		}
320 	}

322 	/**
323 	 * Executes the command.
324 	 * 
325 	 * @return The launched Process.
326 	 * @throws IOException
327 	 *             If an I/O error occurs.
328 	 */
329 	private Process exec() throws IOException {
330 		Runtime rt = Runtime.getRuntime();
331 		Process p;
332 		try {
333 			// java 1.3+
334 			p = rt.exec(command, envs, directory);
335 		} catch (NoSuchMethodError e) {
336 			// java 1.2
337 			p = rt.exec(command, envs);
338 		}
339 		return p;
340 	}

342 	/**
343 	 * Prepares an {@link InputStream} on a file and returns it.
344 	 * 
345 	 * @param file
346 	 *            The file.
347 	 * @return The stream, or null if the file is not found.
348 	 */
349 	private InputStream buildInputStream(File file) {
350 		if (file != null) {
351 			try {
352 				return new FileInputStream(file);
353 			} catch (FileNotFoundException e) {
354 				e.printStackTrace();
355 				return null;
356 			}
357 		}
358 		return null;
359 	}

366 	 * @return The stream, or null if the file is not found.
367 	 */
368 	private OutputStream buildOutputStream(File file) {
369 		if (file != null) {

376 		}
377 		return null;

380 	/**
381 	 * Prints in the returned string the elements contained in the given string
382 	 * array.
383 	 * 
384 	 * @param arr
385 	 *            The array.
386 	 * @return A string representing the supplied array contents.
387 	 */
388 	private static String listStrings(String[] arr) {
389 		if (arr == null) {
390 			return "null";
391 		} else {
392 			StringBuffer b = new StringBuffer();
393 			b.append('[');
394 			for (int i = 0; i < arr.length; i++) {
395 				if (i > 0) {
396 					b.append(", ");
397 				}
398 				b.append(arr[i]);
399 			}
400 			b.append(']');
401 			return b.toString();
402 		}
403 	}

405 	/**
406 	 * Overrides {@link Object#toString()}.
407 	 */
408 	public String toString() {
409 		StringBuffer b = new StringBuffer();
410 		b.append("Task[");
411 		b.append("cmd=");
412 		b.append(ProcessTask.listStrings(command));
413 		b.append(", env=");
414 		b.append(ProcessTask.listStrings(envs));
415 		b.append(", ");
416 		b.append("dir=");
417 		b.append(directory);
418 		b.append("]");
419 		return b.toString();
420 	}
