19  package it.sauronsoftware.cron4j;
20  
21  import java.net.InetAddress;
22  import java.net.NetworkInterface;
23  import java.net.UnknownHostException;
24  import java.util.Enumeration;
25  
26  /**
27   * A GUID generator.
28   * 
29   * @author Carlo Pelliccia
30   * @since 2.0
31   */
32  class GUIDGenerator {
33  
34  	/**
35  	 * The machine descriptor, which is used to identified the underlying hardware machine.
36  	 */
37  	private static String MACHINE_DESCRIPTOR = getMachineDescriptor();
38  	
39  	/**
40  	 * Generates a GUID (48 chars).
41  	 * 
42  	 * @return The generated GUID.
43  	 */
44  	public static String generate() {
45  		StringBuffer id = new StringBuffer();
46  		encode(id, MACHINE_DESCRIPTOR);
47  		encode(id, Runtime.getRuntime());
48  		encode(id, Thread.currentThread());
49  		encode(id, System.currentTimeMillis());
50  		encode(id, getRandomInt());
51  		return id.toString();
52  	}
53  
54  	/**
55  	 * Calculates a machine id, as an integer value.
56  	 * 
57  	 * @return The calculated machine id.
58  	 */
59  	private static String getMachineDescriptor() {
60  		StringBuffer descriptor = new StringBuffer();
61  		descriptor.append(System.getProperty("os.name"));
62  		descriptor.append("::");
63  		descriptor.append(System.getProperty("os.arch"));
64  		descriptor.append("::");
65  		descriptor.append(System.getProperty("os.version"));
66  		descriptor.append("::");
67  		descriptor.append(System.getProperty("user.name"));
68  		descriptor.append("::");
69  		StringBuffer b = buildNetworkInterfaceDescriptor();
70  		if (b != null) {
71  			descriptor.append(b);
72  		} else {
73  			// plain old InetAddress...
74  			InetAddress addr;
75  			try {
76  				addr = InetAddress.getLocalHost();
77  				descriptor.append(addr.getHostAddress());
78  			} catch (UnknownHostException e) {
79  				;
80  			}
81  		}
82  		return descriptor.toString();
83  	}
84  
85  	/**
86  	 * Builds a descriptor fragment using the {@link NetworkInterface} class,
87  	 * available since Java 1.4.
88  	 * 
89  	 * @return A descriptor fragment, or null if the method fails.
90  	 */
91  	private static StringBuffer buildNetworkInterfaceDescriptor() {
92  		Enumeration e1;
93  		try {
94  			e1 = NetworkInterface.getNetworkInterfaces();
95  		} catch (Throwable t) {
96  			// not available
97  			return null;
98  		}
99  		StringBuffer b = new StringBuffer();
100 		while (e1.hasMoreElements()) {
101 			NetworkInterface ni = (NetworkInterface) e1.nextElement();
102 			StringBuffer b1 = getMACAddressDescriptor(ni);
103 			StringBuffer b2 = getInetAddressDescriptor(ni);
104 			StringBuffer b3 = new StringBuffer();
105 			if (b1 != null) {
106 				b3.append(b1);
107 			}
108 			if (b2 != null) {
109 				if (b3.length() > 0) {
110 					b3.append('=');
111 				}
112 				b3.append(b2);
113 			}
114 			if (b3.length() > 0) {
115 				if (b.length() > 0) {
116 					b.append(';');
117 				}
118 				b.append(b3);
119 			}
120 		}
121 		return b;
122 	}
123 
124 	/**
125 	 * Builds a descriptor fragment using the machine MAC address.
126 	 * 
127 	 * @return A descriptor fragment, or null if the method fails.
128 	 */
129 	private static StringBuffer getMACAddressDescriptor(NetworkInterface ni) {
130 		byte[] haddr;
131 		try {
132 			haddr = ni.getHardwareAddress();
133 		} catch (Throwable t) {
134 			// not available.
135 			haddr = null;
136 		}
137 		StringBuffer b = new StringBuffer();
138 		if (haddr != null) {
139 			for (int i = 0; i < haddr.length; i++) {
140 				if (b.length() > 0) {
141 					b.append("-");
142 				}
143 				String hex = Integer.toHexString(0xff & haddr[i]);
144 				if (hex.length() == 1) {
145 					b.append('0');
146 				}
147 				b.append(hex);
148 			}
149 		}
150 		return b;
151 	}
152 
153 	/**
154 	 * Builds a descriptor fragment using the machine inet address.
155 	 * 
156 	 * @return A descriptor fragment, or null if the method fails.
157 	 */
158 	private static StringBuffer getInetAddressDescriptor(NetworkInterface ni) {
159 		StringBuffer b = new StringBuffer();
160 		Enumeration e2 = ni.getInetAddresses();
161 		while (e2.hasMoreElements()) {
162 			InetAddress addr = (InetAddress) e2.nextElement();
163 			if (b.length() > 0) {
164 				b.append(',');
165 			}
166 			b.append(addr.getHostAddress());
167 		}
168 		return b;
169 	}
170 
171 	/**
172 	 * Returns a random integer value.
173 	 * 
174 	 * @return A random integer value.
175 	 */
176 	private static int getRandomInt() {
177 		return (int) Math.round((Math.random() * Integer.MAX_VALUE));
178 	}
179 
180 	/**
181 	 * Encodes an object and appends it to the buffer.
182 	 * 
183 	 * @param b
184 	 *            The buffer.
185 	 * @param obj
186 	 *            The object.
187 	 */
188 	private static void encode(StringBuffer b, Object obj) {
189 		encode(b, obj.hashCode());
190 	}
191 
192 	/**
193 	 * Encodes an integer value and appends it to the buffer.
194 	 * 
195 	 * @param b
196 	 *            The buffer.
197 	 * @param value
198 	 *            The value.
199 	 */
200 	private static void encode(StringBuffer b, int value) {
201 		String hex = Integer.toHexString(value);
202 		int hexSize = hex.length();
203 		for (int i = 8; i > hexSize; i--) {
204 			b.append('0');
205 		}
206 		b.append(hex);
207 	}
208 
209 	/**
210 	 * Encodes a long value and appends it to the buffer.
211 	 * 
212 	 * @param b
213 	 *            The buffer.
214 	 * @param value
215 	 *            The value.
216 	 */
217 	private static void encode(StringBuffer b, long value) {
218 		String hex = Long.toHexString(value);
219 		int hexSize = hex.length();
220 		for (int i = 16; i > hexSize; i--) {
221 			b.append('0');
222 		}
223 		b.append(hex);
224 	}
