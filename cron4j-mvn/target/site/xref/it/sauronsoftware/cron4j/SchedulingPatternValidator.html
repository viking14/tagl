19  package it.sauronsoftware.cron4j;
20  
21  /**
22   * <p>
23   * A scheduling patterns validator.
24   * </p>
25   * 
26   * <p>
27   * The class lets you validate a scheduling pattern before/without using it with
28   * a {@link Scheduler} instance. Simply call:
29   * </p>
30   * 
31   * <pre>
32   * boolean valid = SchedulingPatternValidator.validate(thePattern);
33   * </pre>
34   * 
35   * <p>
36   * It is useful in validating user-entered patterns.
37   * </p>
38   * 
39   * @author Carlo Pelliccia
40   * @deprecated Use {@link SchedulingPattern#validate(String)}.
41   */
42  public class SchedulingPatternValidator {
43  
44  	/**
45  	 * Validates a string as a scheduling pattern.
46  	 * 
47  	 * @param schedulingPattern
48  	 *            The pattern to validate.
49  	 * @return true if the given string represents a valid scheduling pattern;
50  	 *         false otherwise.
51  	 * @deprecated Use {@link SchedulingPattern#validate(String)}.
52  	 */
53  	public static boolean validate(String schedulingPattern) {
54  		return SchedulingPattern.validate(schedulingPattern);
55  	}
56  
57  }
