19  package it.sauronsoftware.cron4j;
20  
21  /**
22   * <p>
23   * A TaskExecutionContext object provides support methods for the execution of a
24   * task. An instance of this class is always passed to the task when its
25   * {@link Task#execute(TaskExecutionContext)} method is called. The task, while
26   * executing, can use the received context to exchange informations with its own
27   * executor. If the task declares to supports pausing, stopping, completeness
28   * tracking and/or status tracking, it has to use its context methods to perform
29   * any declared operation (checks pause and stop requests, sends back tracking
30   * informations).
31   * </p>
32   * 
33   * @author Carlo Pelliccia
34   * @since 2.0
35   */
36  public interface TaskExecutionContext {
37  
38  	/**
39  	 * Returns the scheduler.
40  	 * 
41  	 * @return The scheduler.
42  	 */
43  	public Scheduler getScheduler();
44  
45  	/**
46  	 * Returns the task executor.
47  	 * 
48  	 * @return The task executor.
49  	 */
50  	public TaskExecutor getTaskExecutor();
51  
52  	/**
53  	 * Sets the current status tracking message, that has to be something about
54  	 * what the task is doing at the moment.
55  	 * 
56  	 * @param message
57  	 *            A message representing the current execution status. Null
58  	 *            messages will be blanked.
59  	 */
60  	public void setStatusMessage(String message);
61  
62  	/**
63  	 * Sets the completeness tracking value, that has to be between 0 and 1.
64  	 * 
65  	 * @param completeness
66  	 *            A completeness value, between 0 and 1. Values out of range
67  	 *            will be ignored.
68  	 */
69  	public void setCompleteness(double completeness);
70  
71  	/**
72  	 * If the task execution has been paused, stops until the operation is
73  	 * resumed. It can also returns because of a stop operation without any
74  	 * previous resuming. Due to this the task developer should always check the
75  	 * {@link TaskExecutionContext#isStopped()} value after any
76  	 * <em>pauseIfRequested()</em> call. Note that a task execution can be
77  	 * paused only if the task {@link Task#canBePaused()} method returns
78  	 * <em>true</em>.
79  	 */
80  	public void pauseIfRequested();
81  
82  	/**
83  	 * Checks whether the task execution has been demanded to be stopped. If the
84  	 * returned value is <em>true</em>, the task developer must shut down
85  	 * gracefully its task execution, as soon as possible. Note that a task
86  	 * execution can be stopped only if the task {@link Task#canBePaused()}
87  	 * method returns <em>true</em>.
88  	 * 
89  	 * @return <em>true</em> if the current task execution has been demanded to
90  	 *         be stopped; <em>false</em> otherwise.
91  	 */
92  	public boolean isStopped();
93  
94  }
