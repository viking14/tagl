28  /**
29   * <p>
30   * A UNIX crontab-like pattern is a string split in five space separated parts.
31   * Each part is intented as:
32   * </p>
33   * <ol>
34   * <li><strong>Minutes sub-pattern</strong>. During which minutes of the hour
35   * should the task been launched? The values range is from 0 to 59.</li>
36   * <li><strong>Hours sub-pattern</strong>. During which hours of the day should
37   * the task been launched? The values range is from 0 to 23.</li>
38   * <li><strong>Days of month sub-pattern</strong>. During which days of the
39   * month should the task been launched? The values range is from 1 to 31. The
40   * special value L can be used to recognize the last day of month.</li>
41   * <li><strong>Months sub-pattern</strong>. During which months of the year
42   * should the task been launched? The values range is from 1 (January) to 12
43   * (December), otherwise this sub-pattern allows the aliases &quot;jan&quot;,
44   * &quot;feb&quot;, &quot;mar&quot;, &quot;apr&quot;, &quot;may&quot;,
45   * &quot;jun&quot;, &quot;jul&quot;, &quot;aug&quot;, &quot;sep&quot;,
46   * &quot;oct&quot;, &quot;nov&quot; and &quot;dec&quot;.</li>
47   * <li><strong>Days of week sub-pattern</strong>. During which days of the week
48   * should the task been launched? The values range is from 0 (Sunday) to 6
49   * (Saturday), otherwise this sub-pattern allows the aliases &quot;sun&quot;,
50   * &quot;mon&quot;, &quot;tue&quot;, &quot;wed&quot;, &quot;thu&quot;,
51   * &quot;fri&quot; and &quot;sat&quot;.</li>
52   * </ol>
53   * <p>
54   * The star wildcard character is also admitted, indicating &quot;every minute
55   * of the hour&quot;, &quot;every hour of the day&quot;, &quot;every day of the
56   * month&quot;, &quot;every month of the year&quot; and &quot;every day of the
57   * week&quot;, according to the sub-pattern in which it is used.
58   * </p>
59   * <p>
60   * Once the scheduler is started, a task will be launched when the five parts in
61   * its scheduling pattern will be true at the same time.
62   * </p>
63   * <p>
64   * Some examples:
65   * </p>
66   * <p>
67   * <strong>5 * * * *</strong><br />
68   * This pattern causes a task to be launched once every hour, at the begin of
69   * the fifth minute (00:05, 01:05, 02:05 etc.).
70   * </p>
71   * <p>
72   * <strong>* * * * *</strong><br />
73   * This pattern causes a task to be launched every minute.
74   * </p>
75   * <p>
76   * <strong>* 12 * * Mon</strong><br />
77   * This pattern causes a task to be launched every minute during the 12th hour
78   * of Monday.
79   * </p>
80   * <p>
81   * <strong>* 12 16 * Mon</strong><br />
82   * This pattern causes a task to be launched every minute during the 12th hour
83   * of Monday, 16th, but only if the day is the 16th of the month.
84   * </p>
85   * <p>
86   * Every sub-pattern can contain two or more comma separated values.
87   * </p>
88   * <p>
89   * <strong>59 11 * * 1,2,3,4,5</strong><br />
90   * This pattern causes a task to be launched at 11:59AM on Monday, Tuesday,
91   * Wednesday, Thursday and Friday.
92   * </p>
93   * <p>
94   * Values intervals are admitted and defined using the minus character.
95   * </p>
96   * <p>
97   * <strong>59 11 * * 1-5</strong><br />
98   * This pattern is equivalent to the previous one.
99   * </p>
100  * <p>
101  * The slash character can be used to identify step values within a range. It
102  * can be used both in the form <em>*&#47;c</em> and <em>a-b/c</em>. The
103  * subpattern is matched every <em>c</em> values of the range
104  * <em>0,maxvalue</em> or <em>a-b</em>.
105  * </p>
106  * <p>
107  * <strong>*&#47;5 * * * *</strong><br />
108  * This pattern causes a task to be launched every 5 minutes (0:00, 0:05, 0:10,
109  * 0:15 and so on).
110  * </p>
111  * <p>
112  * <strong>3-18&#47;5 * * * *</strong><br />
113  * This pattern causes a task to be launched every 5 minutes starting from the
114  * third minute of the hour, up to the 18th (0:03, 0:08, 0:13, 0:18, 1:03, 1:08
115  * and so on).
116  * </p>
117  * <p>
118  * <strong>*&#47;15 9-17 * * *</strong><br />
119  * This pattern causes a task to be launched every 15 minutes between the 9th
120  * and 17th hour of the day (9:00, 9:15, 9:30, 9:45 and so on... note that the
121  * last execution will be at 17:45).
122  * </p>
123  * <p>
124  * All the fresh described syntax rules can be used together.
125  * </p>
126  * <p>
127  * <strong>* 12 10-16&#47;2 * *</strong><br />
128  * This pattern causes a task to be launched every minute during the 12th hour
129  * of the day, but only if the day is the 10th, the 12th, the 14th or the 16th
130  * of the month.
131  * </p>
132  * <p>
133  * <strong>* 12 1-15,17,20-25 * *</strong><br />
134  * This pattern causes a task to be launched every minute during the 12th hour
135  * of the day, but the day of the month must be between the 1st and the 15th,
136  * the 20th and the 25, or at least it must be the 17th.
137  * </p>
138  * <p>
139  * Finally cron4j lets you combine more scheduling patterns into one, with the
140  * pipe character:
141  * </p>
142  * <p>
143  * <strong>0 5 * * *|8 10 * * *|22 17 * * *</strong><br />
144  * This pattern causes a task to be launched every day at 05:00, 10:08 and
145  * 17:22.
146  * </p>
147  * 
148  * @author Carlo Pelliccia
149  * @since 2.0
150  */

230 	/**
231 	 * Builds a SchedulingPattern parsing it from a string.
232 	 * 
233 	 * @param pattern
234 	 *            The pattern as a crontab-like string.
235 	 * @throws InvalidPatternException
236 	 *             If the supplied string is not a valid pattern.
237 	 */
238 	public SchedulingPattern(String pattern) throws InvalidPatternException {
239 		this.asString = pattern;
240 		StringTokenizer st1 = new StringTokenizer(pattern, "|");
241 		if (st1.countTokens() < 1) {
242 			throw new InvalidPatternException("invalid pattern: \"" + pattern + "\"");
243 		}
244 		while (st1.hasMoreTokens()) {
245 			String localPattern = st1.nextToken();
246 			StringTokenizer st2 = new StringTokenizer(localPattern, " \t");
247 			if (st2.countTokens() != 5) {
248 				throw new InvalidPatternException("invalid pattern: \"" + localPattern + "\"");
249 			}
250 			try {
251 				minuteMatchers.add(buildValueMatcher(st2.nextToken(), MINUTE_VALUE_PARSER));
252 			} catch (Exception e) {
253 				throw new InvalidPatternException("invalid pattern \""
254 						+ localPattern + "\". Error parsing minutes field: "
255 						+ e.getMessage() + ".");
256 			}
257 			try {
258 				hourMatchers.add(buildValueMatcher(st2.nextToken(), HOUR_VALUE_PARSER));
259 			} catch (Exception e) {
260 				throw new InvalidPatternException("invalid pattern \""
261 						+ localPattern + "\". Error parsing hours field: "
262 						+ e.getMessage() + ".");
263 			}
264 			try {
265 				dayOfMonthMatchers.add(buildValueMatcher(st2.nextToken(), DAY_OF_MONTH_VALUE_PARSER));
266 			} catch (Exception e) {
267 				throw new InvalidPatternException("invalid pattern \""
268 						+ localPattern
269 						+ "\". Error parsing days of month field: "
270 						+ e.getMessage() + ".");
271 			}
272 			try {
273 				monthMatchers.add(buildValueMatcher(st2.nextToken(), MONTH_VALUE_PARSER));
274 			} catch (Exception e) {
275 				throw new InvalidPatternException("invalid pattern \""
276 						+ localPattern + "\". Error parsing months field: "
277 						+ e.getMessage() + ".");
278 			}
279 			try {
280 				dayOfWeekMatchers.add(buildValueMatcher(st2.nextToken(), DAY_OF_WEEK_VALUE_PARSER));
281 			} catch (Exception e) {
282 				throw new InvalidPatternException("invalid pattern \""
283 						+ localPattern
284 						+ "\". Error parsing days of week field: "
285 						+ e.getMessage() + ".");
286 			}
287 			matcherSize++;
288 		}
289 	}

291 	/**
292 	 * A ValueMatcher utility builder.
293 	 * 
294 	 * @param str
295 	 *            The pattern part for the ValueMatcher creation.
296 	 * @param parser
297 	 *            The parser used to parse the values.
298 	 * @return The requested ValueMatcher.
299 	 * @throws Exception
300 	 *             If the supplied pattern part is not valid.
301 	 */
302 	private ValueMatcher buildValueMatcher(String str, ValueParser parser)
303 			throws Exception {
304 		if (str.length() == 1 && str.equals("*")) {
305 			return new AlwaysTrueValueMatcher();
306 		}
307 		ArrayList values = new ArrayList();
308 		StringTokenizer st = new StringTokenizer(str, ",");
309 		while (st.hasMoreTokens()) {
310 			String element = st.nextToken();
311 			ArrayList local;
312 			try {
313 				local = parseListElement(element, parser);
314 			} catch (Exception e) {
315 				throw new Exception("invalid field \"" + str
316 						+ "\", invalid element \"" + element + "\", "
317 						+ e.getMessage());
318 			}
319 			for (Iterator i = local.iterator(); i.hasNext();) {
320 				Object value = i.next();
321 				if (!values.contains(value)) {
322 					values.add(value);
323 				}
324 			}
325 		}
326 		if (values.size() == 0) {
327 			throw new Exception("invalid field \"" + str + "\"");
328 		}
329 		if (parser == DAY_OF_MONTH_VALUE_PARSER) {
330 			return new DayOfMonthValueMatcher(values);
331 		} else {
332 			return new IntArrayValueMatcher(values);
333 		}
334 	}

336 	/**
337 	 * Parses an element of a list of values of the pattern.
338 	 * 
339 	 * @param str
340 	 *            The element string.
341 	 * @param parser
342 	 *            The parser used to parse the values.
343 	 * @return A list of integers representing the allowed values.
344 	 * @throws Exception
345 	 *             If the supplied pattern part is not valid.
346 	 */
347 	private ArrayList parseListElement(String str, ValueParser parser)
348 			throws Exception {
349 		StringTokenizer st = new StringTokenizer(str, "/");
350 		int size = st.countTokens();
351 		if (size < 1 || size > 2) {
352 			throw new Exception("syntax error");
353 		}
354 		ArrayList values;
355 		try {
356 			values = parseRange(st.nextToken(), parser);
357 		} catch (Exception e) {
358 			throw new Exception("invalid range, " + e.getMessage());
359 		}
360 		if (size == 2) {
361 			String dStr = st.nextToken();
362 			int div;
363 			try {
364 				div = Integer.parseInt(dStr);
365 			} catch (NumberFormatException e) {
366 				throw new Exception("invalid divisor \"" + dStr + "\"");
367 			}
368 			if (div < 1) {
369 				throw new Exception("non positive divisor \"" + div + "\"");
370 			}
371 			ArrayList values2 = new ArrayList();
372 			for (int i = 0; i < values.size(); i += div) {
373 				values2.add(values.get(i));
374 			}
375 			return values2;
376 		} else {
377 			return values;
378 		}
379 	}

381 	/**
382 	 * Parses a range of values.
383 	 * 
384 	 * @param str
385 	 *            The range string.
386 	 * @param parser
387 	 *            The parser used to parse the values.
388 	 * @return A list of integers representing the allowed values.
389 	 * @throws Exception
390 	 *             If the supplied pattern part is not valid.
391 	 */
392 	private ArrayList parseRange(String str, ValueParser parser)
393 			throws Exception {
394 		if (str.equals("*")) {
395 			int min = parser.getMinValue();
396 			int max = parser.getMaxValue();
397 			ArrayList values = new ArrayList();
398 			for (int i = min; i <= max; i++) {
399 				values.add(new Integer(i));
400 			}
401 			return values;
402 		}
403 		StringTokenizer st = new StringTokenizer(str, "-");
404 		int size = st.countTokens();
405 		if (size < 1 || size > 2) {
406 			throw new Exception("syntax error");
407 		}
408 		String v1Str = st.nextToken();
409 		int v1;
410 		try {
411 			v1 = parser.parse(v1Str);
412 		} catch (Exception e) {
413 			throw new Exception("invalid value \"" + v1Str + "\", "
414 					+ e.getMessage());
415 		}
416 		if (size == 1) {
417 			ArrayList values = new ArrayList();
418 			values.add(new Integer(v1));
419 			return values;
420 		} else {
421 			String v2Str = st.nextToken();
422 			int v2;
423 			try {
424 				v2 = parser.parse(v2Str);
425 			} catch (Exception e) {
426 				throw new Exception("invalid value \"" + v2Str + "\", "
427 						+ e.getMessage());
428 			}
429 			ArrayList values = new ArrayList();
430 			if (v1 < v2) {
431 				for (int i = v1; i <= v2; i++) {
432 					values.add(new Integer(i));
433 				}
434 			} else if (v1 > v2) {
435 				int min = parser.getMinValue();
436 				int max = parser.getMaxValue();
437 				for (int i = v1; i <= max; i++) {
438 					values.add(new Integer(i));
439 				}
440 				for (int i = min; i <= v2; i++) {
441 					values.add(new Integer(i));
442 				}
443 			} else {
444 				// v1 == v2
445 				values.add(new Integer(v1));
446 			}
447 			return values;
448 		}
449 	}

451 	/**
452 	 * This methods returns true if the given timestamp (expressed as a UNIX-era
453 	 * millis value) matches the pattern, according to the given time zone.
454 	 * 
455 	 * @param timezone
456 	 *            A time zone.
457 	 * @param millis
458 	 *            The timestamp, as a UNIX-era millis value.
459 	 * @return true if the given timestamp matches the pattern.
460 	 */
461 	public boolean match(TimeZone timezone, long millis) {
462 		GregorianCalendar gc = new GregorianCalendar();
463 		gc.setTimeInMillis(millis);
464 		gc.setTimeZone(timezone);
465 		int minute = gc.get(Calendar.MINUTE);
466 		int hour = gc.get(Calendar.HOUR_OF_DAY);
467 		int dayOfMonth = gc.get(Calendar.DAY_OF_MONTH);
468 		int month = gc.get(Calendar.MONTH) + 1;
469 		int dayOfWeek = gc.get(Calendar.DAY_OF_WEEK) - 1;
470 		int year = gc.get(Calendar.YEAR);
471 		for (int i = 0; i < matcherSize; i++) {
472 			ValueMatcher minuteMatcher = (ValueMatcher) minuteMatchers.get(i);
473 			ValueMatcher hourMatcher = (ValueMatcher) hourMatchers.get(i);
474 			ValueMatcher dayOfMonthMatcher = (ValueMatcher) dayOfMonthMatchers.get(i);
475 			ValueMatcher monthMatcher = (ValueMatcher) monthMatchers.get(i);
476 			ValueMatcher dayOfWeekMatcher = (ValueMatcher) dayOfWeekMatchers.get(i);
477 			boolean eval = minuteMatcher.match(minute)
478 					&& hourMatcher.match(hour)
479 					&& ((dayOfMonthMatcher instanceof DayOfMonthValueMatcher) ? ((DayOfMonthValueMatcher) dayOfMonthMatcher)
480 							.match(dayOfMonth, month, gc.isLeapYear(year))
481 							: dayOfMonthMatcher.match(dayOfMonth))
482 					&& monthMatcher.match(month)
483 					&& dayOfWeekMatcher.match(dayOfWeek);
484 			if (eval) {
485 				return true;
486 			}
487 		}
488 		return false;
489 	}

491 	/**
492 	 * This methods returns true if the given timestamp (expressed as a UNIX-era
493 	 * millis value) matches the pattern, according to the system default time
494 	 * zone.
495 	 * 
496 	 * @param millis
497 	 *            The timestamp, as a UNIX-era millis value.
498 	 * @return true if the given timestamp matches the pattern.
499 	 */
500 	public boolean match(long millis) {
501 		return match(TimeZone.getDefault(), millis);
502 	}

514 	 * This utility method changes an alias to an int value.
515 	 * 
516 	 * @param value
517 	 *            The value.
518 	 * @param aliases
519 	 *            The aliases list.
520 	 * @param offset
521 	 *            The offset appplied to the aliases list indices.
522 	 * @return The parsed value.
523 	 * @throws Exception
524 	 *             If the expressed values doesn't match any alias.
525 	 */
526 	private static int parseAlias(String value, String[] aliases, int offset)
527 			throws Exception {
528 		for (int i = 0; i < aliases.length; i++) {
529 			if (aliases[i].equalsIgnoreCase(value)) {
530 				return offset + i;
531 			}
532 		}
533 		throw new Exception("invalid alias \"" + value + "\"");
534 	}

568 	/**
569 	 * A simple value parser.
570 	 */
571 	private static class SimpleValueParser implements ValueParser {
572 
573 		/**
574 		 * The minimum allowed value.
575 		 */
576 		protected int minValue;
577 
578 		/**
579 		 * The maximum allowed value.
580 		 */
581 		protected int maxValue;
582 
583 		/**
584 		 * Builds the value parser.
585 		 * 
586 		 * @param minValue
587 		 *            The minimum allowed value.
588 		 * @param maxValue
589 		 *            The maximum allowed value.
590 		 */
591 		public SimpleValueParser(int minValue, int maxValue) {
592 			this.minValue = minValue;
593 			this.maxValue = maxValue;
594 		}
595 
596 		public int parse(String value) throws Exception {
597 			int i;
598 			try {
599 				i = Integer.parseInt(value);
600 			} catch (NumberFormatException e) {
601 				throw new Exception("invalid integer value");
602 			}
603 			if (i < minValue || i > maxValue) {
604 				throw new Exception("value out of range");
605 			}
606 			return i;
607 		}
608 
609 		public int getMinValue() {
610 			return minValue;
611 		}
612 
613 		public int getMaxValue() {
614 			return maxValue;
615 		}
616 
617 	}

647 	/**
648 	 * The days of month value parser.
649 	 */
650 	private static class DayOfMonthValueParser extends SimpleValueParser {
651 
652 		/**
653 		 * Builds the value parser.
654 		 */
655 		public DayOfMonthValueParser() {
656 			super(1, 31);
657 		}
658 
659 		/**
660 		 * Added to support last-day-of-month.
661 		 * 
662 		 * @param value
663 		 *            The value to be parsed
664 		 * @return the integer day of the month or 32 for last day of the month
665 		 * @throws Exception
666 		 *             if the input value is invalid
667 		 */
