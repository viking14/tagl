19  package it.sauronsoftware.cron4j;
20  
21  /**
22   * <p>
23   * This interface describes a task collector. Task collectors can be registered
24   * in a {@link Scheduler} instance with the
25   * {@link Scheduler#addTaskCollector(TaskCollector)} method. Any registered task
26   * collector is queried by the scheduler once a minute. The developer has to
27   * implement the {@link TaskCollector#getTasks()} method, returning a
28   * {@link TaskTable} whose elements has been collected with a custom logic. In
29   * example the list can be extracted from a database.
30   * </p>
31   * 
32   * @author Carlo Pelliccia
33   * @since 2.0
34   */
35  public interface TaskCollector {
36  
37  	/**
38  	 * Once the instance has been registered on a {@link Scheduler} instance,
39  	 * with the {@link Scheduler#addTaskCollector(TaskCollector)} method, this
40  	 * method will be queried once a minute. It should return a custom
41  	 * {@link TaskTable} object. The scheduler instance will automatically
42  	 * iterate over the returned table elements, executing any task whose
43  	 * scheduling pattern is matching the current system time.
44  	 * 
45  	 * @return The task table that will be automatically injected in the
46  	 *         scheduler.
47  	 */
48  	public TaskTable getTasks();
49  
50  }
