32  /**
33   * <p>
34   * A parser for crontab-like formatted files and streams.
35   * </p>
36   * <p>
37   * If you want to schedule a list of tasks declared in a crontab-like file you
38   * don't need the CronParser, since you can do it by adding the file to the
39   * scheduler, with the {@link Scheduler#scheduleFile(File)} method.
40   * </p>
41   * <p>
42   * Consider to use the CronParser if the {@link Scheduler#scheduleFile(File)}
43   * method is not enough for you. In example, you may need to fetch the task list
44   * from a remote source which is not representable as a {@link File} object (a
45   * document on a remote server, a DBMS result set and so on). To solve the
46   * problem you can implement your own {@link TaskCollector}, getting the
47   * advantage of the CronParser to parse easily any crontab-like content.
48   * </p>
49   * <p>
50   * You can parse a whole file/stream, but you can also parse a single line.
51   * </p>
52   * <p>
53   * A line can be empty, can contain a comment or it can be a scheduling line.
54   * </p>
55   * <p>
56   * A line containing no characters or a line with only space characters is
57   * considered an empty line.
58   * </p>
59   * <p>
60   * A line whose first non-space character is a number sign (#) is considered a
61   * comment.
62   * </p>
63   * <p>
64   * Empty lines and comment lines are ignored by the parser.
65   * </p>
66   * <p>
67   * Any other kind of line is parsed as a scheduling line.
68   * </p>
69   * <p>
70   * A valid scheduling line respects the following structure:
71   * </p>
72   * 
73   * <pre>
74   * scheduling-pattern [options] command [args]
75   * </pre>
76   * 
77   * <ul>
78   * <li><em>scheduling-pattern</em> is a valid scheduling pattern, according with
79   * the definition given by the {@link SchedulingPattern} class.</li>
80   * <li><em>options</em> is a list of optional informations used by cron4j to
81   * prepare the task execution environment. See below for a more detailed
82   * description.</li>
83   * <li><em>command</em> is a system valid command, such an executable call.</li>
84   * <li><em>args</em> is a list of optional arguments for the command.</li>
85   * </ul>
86   * <p>
87   * After the scheduling pattern item, other tokens in each line are space
88   * separated or delimited with double quotation marks (&quot;).
89   * </p>
90   * <p>
91   * Double quotation marks delimited items can take advantage of the following
92   * escape sequences:
93   * </p>
94   * <ul>
95   * <li>\&quot; - quotation mark</li>
96   * <li>\\ - back slash</li>
97   * <li>\/ - slash</li>
98   * <li>\b - back space</li>
99   * <li>\f - form feed</li>
100  * <li>\n - new line</li>
101  * <li>\r - carriage return</li>
102  * <li>\t - horizontal tab</li>
103  * <li>\u005c\u0075<em>four-hex-digits</em> - the character at the given unicode
104  * index</li>
105  * </ul>
106  * <p>
107  * The <em>options</em> token collection can include one or more of the
108  * following elements:
109  * </p>
110  * <ul>
111  * <li>IN:<em>file-path</em> - Redirects the command standard input channel to
112  * the specified file.</li>
113  * <li>OUT:<em>file-path</em> - Redirects the command standard output channel to
114  * the specified file.</li>
115  * <li>ERR:<em>file-path</em> - Redirects the command standard error channel to
116  * the specified file.</li>
117  * <li>ENV:<em>name</em>=<em>value</em> - Defines an environment variable in the
118  * scope of the command.</li>
119  * <li>DIR:<em>directory-path</em> - Sets the path of the working directory for
120  * the command. This feature is not supported if the executing JVM is less than
121  * 1.3.</li>
122  * </ul>
123  * <p>
124  * It is also possible to schedule the invocation of a method of a Java class in
125  * the scope of the parser ClassLoader. The method has to be static and it must
126  * accept an array of strings as its sole argument. To invoke a method of this
127  * kind the syntax is:
128  * </p>
129  * 
130  * <pre>
131  * scheduling-pattern java:className#methodName [args]
132  * </pre>
133  * <p>
134  * The <em>#methodName</em> part can be omitted: in this case the
135  * <em>main(String[])</em> method will be assumed.
136  * </p>
137  * <p>
138  * Please note that static methods are invoked within the scheduler same JVM,
139  * without spawning any external process. Thus IN, OUT, ERR, ENV and DIR options
140  * can't be applied.
141  * </p>
142  * <p>
143  * Invalid scheduling lines are discarded without blocking the parsing
144  * procedure, but an error message is printed in the application standard error
145  * channel.
146  * </p>
147  * <p>
148  * Valid examples:
149  * </p>
150  * 
151  * <pre>
152  * 0 5 * * * sol.exe
153  * 0,30 * * * * OUT:C:\ping.txt ping 10.9.43.55
154  * 0,30 4 * * * &quot;OUT:C:\Documents and Settings\Carlo\ping.txt&quot; ping 10.9.43.55
155  * 0 3 * * * ENV:JAVA_HOME=C:\jdks\1.4.2_15 DIR:C:\myproject OUT:C:\myproject\build.log C:\myproject\build.bat &quot;Nightly Build&quot;
156  * 0 4 * * * java:mypackage.MyClass#startApplication myOption1 myOption2
157  * </pre>
158  * 
159  * @author Carlo Pelliccia
160  * @since 2.0
161  */
162 public class CronParser {

166 	 */

170 	/**
171 	 * <p>
172 	 * Builds a task list reading it from a file.
173 	 * </p>
174 	 * 
175 	 * <p>
176 	 * The file is treated as UTF-8. If your source file is not UTF-8 encoded
177 	 * establish by yourself a {@link Reader} using the right charset and pass
178 	 * it to the {@link CronParser#parse(Reader)} method.
179 	 * </p>
180 	 * 
181 	 * <p>
182 	 * Syntax and semantics errors in the source file are not blocking. Invalid
183 	 * lines are discarded, and they cause just a stack trace to be printed in
184 	 * the standard error channel as a notification.
185 	 * </p>
186 	 * 
187 	 * @param file
188 	 *            The file.
189 	 * @return The task table parsed from the file.
190 	 * @throws IOException
191 	 *             I/O error.
192 	 */
193 	public static TaskTable parse(File file) throws IOException {
194 		InputStream stream = null;
195 		try {
196 			stream = new FileInputStream(file);
197 			return parse(stream);
198 		} finally {
199 			if (stream != null) {
200 				try {
201 					stream.close();
202 				} catch (Throwable t) {
203 					;
204 				}
205 			}
206 		}
207 	}

209 	/**
210 	 * <p>
211 	 * Builds a task list reading it from an URL.
212 	 * </p>
213 	 * 
214 	 * <p>
215 	 * Contents fetched from the URL are treated as UTF-8. If your source is not
216 	 * UTF-8 encoded establish by yourself a {@link Reader} using the right
217 	 * charset and pass it to the {@link CronParser#parse(Reader)} method.
218 	 * </p>
219 	 * 
220 	 * <p>
221 	 * Syntax and semantics errors in the retrieved document are not blocking.
222 	 * Invalid lines are discarded, and they cause just a stack trace to be
223 	 * printed in the standard error channel as a notification.
224 	 * </p>
225 	 * 
226 	 * @param url
227 	 *            The URL.
228 	 * @return The task table parsed from the contents fetched from the given
229 	 *         URL.
230 	 * @throws IOException
231 	 *             I/O error.
232 	 */
233 	public static TaskTable parse(URL url) throws IOException {
234 		InputStream stream = null;
235 		try {
236 			stream = url.openStream();
237 			return parse(stream);
238 		} finally {
239 			if (stream != null) {
240 				try {
241 					stream.close();
242 				} catch (Throwable t) {
243 					;
244 				}
245 			}
246 		}
247 	}

249 	/**
250 	 * <p>
251 	 * Builds a task list reading it from an input stream.
252 	 * </p>
253 	 * 
254 	 * <p>
255 	 * The stream is treated as UTF-8. If your source is not UTF-8 encoded
256 	 * establish by yourself a {@link Reader} using the right charset and pass
257 	 * it to the {@link CronParser#parse(Reader)} method.
258 	 * </p>
259 	 * 
260 	 * <p>
261 	 * Syntax and semantics errors in the source stream are not blocking.
262 	 * Invalid lines are discarded, and they cause just a stack trace to be
263 	 * printed in the standard error channel as a notification.
264 	 * </p>
265 	 * 
266 	 * @param stream
267 	 *            The input stream.
268 	 * @return The task table parsed from the stream contents.
269 	 * @throws IOException
270 	 *             I/O error.
271 	 */
272 	public static TaskTable parse(InputStream stream) throws IOException {
273 		return parse(new InputStreamReader(stream, "UTF-8"));
274 	}

276 	/**
277 	 * <p>
278 	 * Builds a task list reading it from a reader.
279 	 * </p>
280 	 * 
281 	 * <p>
282 	 * Syntax and semantics errors in the source reader are not blocking.
283 	 * Invalid lines are discarded, and they cause just a stack trace to be
284 	 * printed in the standard error channel as a notification.
285 	 * </p>
286 	 * 
287 	 * @param reader
288 	 *            The reader.
289 	 * @return The task table parsed from the contents in the reader.
290 	 * @throws IOException
291 	 *             I/O error.
292 	 */
293 	public static TaskTable parse(Reader reader) throws IOException {
294 		TaskTable table = new TaskTable();
295 		BufferedReader bufferedReader = new BufferedReader(reader);
296 		try {
297 			String line;
298 			while ((line = bufferedReader.readLine()) != null) {
299 				try {
300 					parseLine(table, line);
301 				} catch (Exception e) {
302 					e.printStackTrace();
303 					continue;
304 				}
305 			}
306 		} finally {
307 			reader.close();
308 		}
309 		return table;
310 	}

312 	/**
313 	 * Parses a crontab-like line.
314 	 * 
315 	 * @param table
316 	 *            The table on which the parsed task will be stored, by
317 	 *            side-effect.
318 	 * @param line
319 	 *            The crontab-like line.
320 	 * @throws Exception
321 	 *             The supplied line doesn't represent a valid task line.
322 	 */
323 	public static void parseLine(TaskTable table, String line) throws Exception {
324 		line = line.trim();
325 		if (line.length() == 0 || line.charAt(0) == '#') {
326 			return;
327 		}
328 		// Detecting the pattern.
329 		int size = line.length();
330 		String pattern = null;
331 		for (int i = size; i >= 0; i--) {
332 			String aux = line.substring(0, i);
333 			if (SchedulingPattern.validate(aux)) {
334 				pattern = aux;
335 				break;
336 			}
337 		}
338 		if (pattern == null) {
339 			throw new Exception("Invalid cron line: " + line);
340 		}
341 		line = line.substring(pattern.length());
342 		size = line.length();
343 		// Splitting the line
344 		ArrayList splitted = new ArrayList();
345 		StringBuffer current = null;
346 		boolean quotes = false;
347 		for (int i = 0; i < size; i++) {
348 			char c = line.charAt(i);
349 			if (current == null) {
350 				if (c == '"') {
351 					current = new StringBuffer();
352 					quotes = true;
353 				} else if (c > ' ') {
354 					current = new StringBuffer();
355 					current.append(c);
356 					quotes = false;
357 				}
358 			} else {
359 				boolean closeCurrent;
360 				if (quotes) {
361 					closeCurrent = (c == '"');
362 				} else {
363 					closeCurrent = (c <= ' ');
364 				}
365 				if (closeCurrent) {
366 					if (current != null && current.length() > 0) {
367 						String str = current.toString();
368 						if (quotes) {
369 							str = escape(str);
370 						}
371 						splitted.add(str);
372 					}
373 					current = null;
374 				} else {
375 					current.append(c);
376 				}
377 			}
378 		}
379 		if (current != null && current.length() > 0) {
380 			String str = current.toString();
381 			if (quotes) {
382 				str = escape(str);
383 			}
384 			splitted.add(str);
385 			current = null;
386 		}
387 		// Analyzing
388 		size = splitted.size();
389 		int status = 0;
390 		// Status values:
391 		// 0 -> fetching environment variables, working directory and channels
392 		// 1 -> fetching the command and its arguments
393 		String dirString = null;
394 		File stdinFile = null;
395 		File stdoutFile = null;
396 		File stderrFile = null;
397 		ArrayList envsList = new ArrayList();
398 		String command = null;
399 		ArrayList argsList = new ArrayList();
400 		for (int i = 0; i < size; i++) {
401 			String tk = (String) splitted.get(i);
402 			// Check the local status.
403 			if (status == 0) {
404 				// Environment variables, working directory and channels
405 				if (tk.startsWith("ENV:")) {
406 					envsList.add(tk.substring(4));
407 					continue;
408 				} else if (tk.startsWith("DIR:")) {
409 					dirString = tk.substring(4);
410 					continue;
411 				} else if (tk.startsWith("IN:")) {
412 					stdinFile = new File(tk.substring(3));
413 					continue;
414 				} else if (tk.startsWith("OUT:")) {
415 					stdoutFile = new File(tk.substring(4));
416 					continue;
417 				} else if (tk.startsWith("ERR:")) {
418 					stderrFile = new File(tk.substring(4));
419 					continue;
420 				} else {
421 					status = 1;
422 				}
423 			}
424 			if (status == 1) {
425 				// Command or argument?
426 				if (command == null) {
427 					command = tk;
428 				} else {
429 					argsList.add(tk);
430 				}
431 			}
432 		}
433 		// Task preparing.
434 		Task task;
435 		// Command evaluation.
436 		if (command == null) {
437 			// No command!
438 			throw new Exception("Invalid cron line: " + line);
439 		} else if (command.startsWith("java:")) {
440 			// Java inner-process.
441 			String className = command.substring(5);
442 			if (className.length() == 0) {
443 				throw new Exception("Invalid Java class name on line: " + line);
444 			}
445 			String methodName;
446 			int sep = className.indexOf('#');
447 			if (sep == -1) {
448 				methodName = "main";
449 			} else {
450 				methodName = className.substring(sep + 1);
451 				className = className.substring(0, sep);
452 				if (methodName.length() == 0) {
453 					throw new Exception("Invalid Java method name on line: "
454 							+ line);
455 				}
456 			}
457 			String[] args = new String[argsList.size()];
458 			for (int i = 0; i < argsList.size(); i++) {
459 				args[i] = (String) argsList.get(i);
460 			}
461 			task = new StaticMethodTask(className, methodName, args);
462 		} else {
463 			// External command.
464 			String[] cmdarray = new String[1 + argsList.size()];
465 			cmdarray[0] = command;
466 			for (int i = 0; i < argsList.size(); i++) {
467 				cmdarray[i + 1] = (String) argsList.get(i);
468 			}
469 			// Environments.
470 			String[] envs = null;
471 			size = envsList.size();
472 			if (size > 0) {
473 				envs = new String[size];
474 				for (int i = 0; i < size; i++) {
475 					envs[i] = (String) envsList.get(i);
476 				}
477 			}
478 			// Working directory.
479 			File dir = null;
480 			if (dirString != null) {
481 				dir = new File(dirString);
482 				if (!dir.exists() || !dir.isDirectory()) {
483 					throw new Exception(
484 							"Invalid cron working directory parameter at line: "
485 									+ line,
486 							new FileNotFoundException(dirString
487 									+ " doesn't exist or it is not a directory"));
488 				}
489 			}
490 			// Builds the task.
491 			ProcessTask process = new ProcessTask(cmdarray, envs, dir);
492 			// Channels.
493 			if (stdinFile != null) {
494 				process.setStdinFile(stdinFile);
495 			}
496 			if (stdoutFile != null) {
497 				process.setStdoutFile(stdoutFile);
498 			}
499 			if (stderrFile != null) {
500 				process.setStderrFile(stderrFile);
501 			}
502 			task = process;
503 		}
504 		// End.
505 		table.add(new SchedulingPattern(pattern), task);
506 	}

508 	/**
509 	 * Escapes special chars occurrences.
510 	 * 
511 	 * @param str
512 	 *            The input stream.
513 	 * @return The decoded output stream.
514 	 */
515 	private static String escape(String str) {
516 		int size = str.length();
517 		StringBuffer b = new StringBuffer();
518 		for (int i = 0; i < size; i++) {
519 			int skip = 0;
520 			char c = str.charAt(i);
521 			if (c == '\\') {
522 				if (i < size - 1) {
523 					char d = str.charAt(i + 1);
524 					if (d == '"') {
525 						b.append('"');
526 						skip = 2;
527 					} else if (d == '\\') {
528 						b.append('\\');
529 						skip = 2;
530 					} else if (d == '/') {
531 						b.append('/');
532 						skip = 2;
533 					} else if (d == 'b') {
534 						b.append('\b');
535 						skip = 2;
536 					} else if (d == 'f') {
537 						b.append('\f');
538 						skip = 2;
539 					} else if (d == 'n') {
540 						b.append('\n');
541 						skip = 2;
542 					} else if (d == 'r') {
543 						b.append('\r');
544 						skip = 2;
545 					} else if (d == 't') {
546 						b.append('\t');
547 						skip = 2;
548 					} else if (d == 'u') {
549 						if (i < size - 5) {
550 							String hex = str.substring(i + 2, i + 6);
551 							try {
552 								int code = Integer.parseInt(hex, 16);
553 								if (code >= 0) {
554 									b.append((char) code);
555 									skip = 6;
556 								}
557 							} catch (NumberFormatException e) {
558 								;
559 							}
560 						}
561 					}
562 				}
563 			}
564 			if (skip == 0) {
565 				b.append(c);
566 			} else {
567 				i += (skip - 1);
568 			}
569 		}
570 		return b.toString();
571 	}
