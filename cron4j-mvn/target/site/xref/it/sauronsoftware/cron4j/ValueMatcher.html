19  package it.sauronsoftware.cron4j;

21  /**
22   * <p>
23   * This interface describes the ValueMatcher behavior. A ValueMatcher is an
24   * object that validate an integer value against a set of rules.
25   * </p>
26   * 
27   * @author Carlo Pelliccia
28   */
29  interface ValueMatcher {
30  
31  	/**
32  	 * Validate the given integer value against a set of rules.
33  	 * 
34  	 * @param value
35  	 *            The value.
36  	 * @return true if the given value matches the rules of the ValueMatcher,
37  	 *         false otherwise.
38  	 */
39  	public boolean match(int value);
40  
41  }
