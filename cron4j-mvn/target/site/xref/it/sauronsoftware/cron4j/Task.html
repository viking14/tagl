19  package it.sauronsoftware.cron4j;
20  
21  /**
22   * <p>
23   * Abstract base representation of a cron4j task.
24   * </p>
25   * <p>
26   * Developers can extends this abstract class to build their own tasks.
27   * </p>
28   * <p>
29   * Extending Task means, above all, implementing the
30   * {@link Task#execute(TaskExecutionContext)} method. Within this method the
31   * task must perform its operation. If the <em>execute()</em> method returns
32   * regularly then the execution is considered to be successfully completed. If
33   * <em>execute()</em> dies throwing a {@link RuntimeException} then the task
34   * execution is considered to be failed. The supplied parameter, which is a
35   * {@link TaskExecutionContext} instance, helps the developer in integrating his
36   * task with the scheduler executor. Through the context the developer can check
37   * if the execution has been paused or stopped, and he can also push back some
38   * status informations by calling
39   * {@link TaskExecutionContext#setCompleteness(double)} and
40   * {@link TaskExecutionContext#setStatusMessage(String)}.
41   * </p>
42   * <p>
43   * If the custom task supports pausing, stopping and/or tracking, that should be
44   * notified by overriding {@link Task#canBePaused()},
45   * {@link Task#canBeStopped()}, {@link Task#supportsCompletenessTracking()}
46   * and/or {@link Task#supportsStatusTracking()}.
47   * </p>
48   * 
49   * @author Carlo Pelliccia
50   * @since 2.0
51   */
52  public abstract class Task {
53  
54  	/**
55  	 * The ID for this task. Also used as an instance synchronization lock.
56  	 */
57  	private Object id = GUIDGenerator.generate();
58  
59  	/**
60  	 * Empty constructor, does nothing.
61  	 */
62  	public Task() {
63  	}
64  
65  	/**
66  	 * It returns the ID for this task.
67  	 * 
68  	 * @return The ID for this task.
69  	 */
70  	Object getId() {
71  		return id;
72  	}
73  
74  	/**
75  	 * <p>
76  	 * Checks whether this task supports pause requests.
77  	 * </p>
78  	 * <p>
79  	 * Default implementation returns <em>false</em>.
80  	 * </p>
81  	 * <p>
82  	 * Task developers can override this method to let it return a <em>true</em>
83  	 * value, and at the same time they have to implement the
84  	 * {@link Task#execute(TaskExecutionContext)} method, so that pause requests
85  	 * are really handled. This can be done calling regularly the
86  	 * {@link TaskExecutionContext#pauseIfRequested()} method during the task
87  	 * execution.
88  	 * </p>
89  	 * 
90  	 * @return true if this task can be paused; false otherwise.
91  	 */
92  	public boolean canBePaused() {
93  		return false;
94  	}
95  
96  	/**
97  	 * <p>
98  	 * Checks whether this task supports stop requests.
99  	 * </p>
100 	 * <p>
101 	 * Default implementation returns <em>false</em>.
102 	 * </p>
103 	 * <p>
104 	 * Task developers can override this method to let it return a <em>true</em>
105 	 * value, and at the same time they have to implement the
106 	 * {@link Task#execute(TaskExecutionContext)} method, so that stop requests
107 	 * are really handled. This can be done checking regularly the
108 	 * {@link TaskExecutionContext#isStopped()} method during the task
109 	 * execution.
110 	 * </p>
111 	 * 
112 	 * @return true if this task can be stopped; false otherwise.
113 	 */
114 	public boolean canBeStopped() {
115 		return false;
116 	}
117 
118 	/**
119 	 * <p>
120 	 * Tests whether this task supports status tracking.
121 	 * </p>
122 	 * <p>
123 	 * Default implementation returns <em>false</em>.
124 	 * </p>
125 	 * <p>
126 	 * The task developer can override this method and returns <em>true</em>,
127 	 * having care to regularly calling the
128 	 * {@link TaskExecutionContext#setStatusMessage(String)} method during the
129 	 * task execution.
130 	 * </p>
131 	 * 
132 	 * @return true if this task, during its execution, provides status message
133 	 *         regularly.
134 	 */
135 	public boolean supportsStatusTracking() {
136 		return false;
137 	}
138 
139 	/**
140 	 * <p>
141 	 * Tests whether this task supports completeness tracking.
142 	 * </p>
143 	 * <p>
144 	 * Default implementation returns <em>false</em>.
145 	 * </p>
146 	 * <p>
147 	 * The task developer can override this method and returns <em>true</em>,
148 	 * having care to regularly calling the
149 	 * {@link TaskExecutionContext#setCompleteness(double)} method during the
150 	 * task execution.
151 	 * </p>
152 	 * 
153 	 * @return true if this task, during its execution, provides a completeness
154 	 *         value regularly.
155 	 */
156 	public boolean supportsCompletenessTracking() {
157 		return false;
158 	}
159 
160 	/**
161 	 * <p>
162 	 * This method is called to require a task execution, and should contain the
163 	 * core routine of any scheduled task.
164 	 * </p>
165 	 * 
166 	 * <p>
167 	 * If the <em>execute()</em> method ends regularly the scheduler will
168 	 * consider the execution successfully completed, and this will be
169 	 * communicated to any {@link SchedulerListener} interested in it. If the
170 	 * <em>execute()</em> method dies throwing a {@link RuntimeException} the
171 	 * scheduler will consider it as a failure notification. Any
172 	 * {@link SchedulerListener} will be notified about the occurred exception.
173 	 * </p>
174 	 * 
175 	 * @param context
176 	 *            The execution context.
177 	 * @throws RuntimeException
178 	 *             Task execution has somehow failed.
179 	 */
180 	public abstract void execute(TaskExecutionContext context)
181 			throws RuntimeException;
182 
183 }
